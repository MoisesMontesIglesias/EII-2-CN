def biseccion(f,aa,bb):
    # Solo aplica bisección si se cumplen las condiciones necesarias
    if f(aa)*f(bb) < 0:
        cond = True
        while cond == True:
            m = (aa+bb)/2.
            if f(aa)*f(m) < 0:
                bb = m
            elif f(m)*f(bb) < 0:
                aa = m
            else:
                return (True,m)
            cond = abs(bb-aa)/2. > 1.e-12
        return (True,m)   
    # En caso contrario devuelve un False
    else:
        return (False,a)

def raices(f,a,b):
    '''
    - Dividimos [a,b] en intervalos de longitud dx
    y los enviamos a la función bisección
    si no se cumplen las condiciones para aplicar bisección
    no hace nada y pasa al intervalo siguiente.
    
    - Caso contrario la función la función almacena 
    la solución obtenida por bisección
    '''
    sol = []
    dx = 0.05
    valores = np.arange(a,b,dx)

    for aa in valores:
        bb = aa + dx
        sal = biseccion(f,aa,bb)
        if sal[0] == True:
            sol.append(sal[1])
    return sol

def dif_div(x,y):
    n = len(x)
    d = np.zeros((n,n))
    d[:,0] = y
    for j in range(1,n):
        for i in range(n-j):
            d[i,j] = d[i+1,j-1]-d[i,j-1]
            d[i,j] /= (x[i+j]-x[i])
    return d

def newton_polinomio(z,x,y):
    n = len(x)
    d = dif_div(x,y)

    P = 0.
    factor = 1.
    for i in range(n): 
        P += d[0,i]*factor
        factor *= (z-x[i])
    return P

Empezaremos linealizando la función a ajustar. Se cumple que,

$$Q_{k}=a\:e^{\,b\:t_{k}}\Longrightarrow\ln Q_{k}=\ln\left(a\:e^{\,b\:t_k}\right)\Longrightarrow\ln Q_{k}=\ln a+\ln e^{b\:t}\Longrightarrow\ln Q_{k}=\ln a+\left(b\:t_{k}\right)\ln e\Longrightarrow\ln Q_{k}=\ln a+b\:t_{k}$$

y si llamamos

$$y_{k}=\ln Q_{k},\quad x_{k}=t_{k},\quad a_{0}=\ln a,\quad a_{1}=b$$

tenemos que

$$\ln Q_{k}=\ln a+b\:t_{k}\quad\Longrightarrow\quad y_{k}=a_{0}+a_{1}x_{k}$$

y el problema es ahora ajustar una recta de regresión mínimo cuadrática

$$P_{1}(x)=a_{0}+a_{1}\,x$$

que se resuelve con el sistema

$$ \left(\begin{array}{cc} \sum_{k=1}^{6}1 & \sum_{k=1}^{6}x_{k}\\ \sum_{k=1}^{6}x_{k} & \sum_{k=1}^{6}x_{k}^{2} \end{array}\right)\left(\begin{array}{c} a_{0}\\ a_{1} \end{array}\right)=\left(\begin{array}{c} \sum_{k=1}^{6}y_{k}\\ \sum_{k=1}^{6}x_{k}y_{k} \end{array}\right) $$

def ajusta_exp(t,Q):
    x = t
    y = np.log(Q)

    A = np.array([[len(x),sum(x)],[sum(x),sum(x**2)]])
    b = np.array([sum(y),sum(x*y)])

    a = np.linalg.solve(A,b) 
    
    aa = np.exp(a[0])
    b = a[1]
    tt = np.linspace(min(t),max(t))
    QQ = lambda t: aa*np.exp(b*t)
    plt.plot(tt,QQ(tt),t,Q,'ro')
    plt.show()
    
    return aa,b

def gauss_leg(f,a,b,n):
    [x, w] = np.polynomial.legendre.leggauss(n)
    y = (b-a)/2*x+(b+a)/2
    I = sum(w*f(y))*(b-a)/2

    xx = np.linspace(a,b)
    plt.plot(xx,f(xx),label=u'función')
    plt.plot(y,f(y),'ro',label='nodos')

    pol = np.polyfit(y,f(y),len(y)-1)
    P = np.polyval(pol,xx)
    plt.plot(xx,P,'r-',label='polinomio')
    plt.legend()
    plt.show()
    return I
